C# Код для запуску трансформації

using System;
using System.Xml.Xsl;

static void Main(string[] args) 
{
        XslCompiledTransform xsltCodeGnerator = new XslCompiledTransform();
        xsltCodeGnerator.Load("Fields.xslt");
        xsltCodeGnerator.Transform("Fields.xml", "Fields.cs");
}
